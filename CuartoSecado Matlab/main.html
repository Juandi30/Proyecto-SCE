Función de Transferencia obtenida (Metodo 3)

%% Función de Transferencia obtenida (Metodo 3)
%G=tf(d2c(tf9));%<---Modelo con mejor FIT
G=tf(tf9);%<---Modelo con mejor FIT
num=cell2mat(G.numerator);
den=cell2mat(G.denominator);
FTTemp=tf(num,den)
FTTemp = 0.002885 s^6 - 0.02553 s^5 + 0.001692 s^4 - 0.005553 s^3 + 0.000189 s^2 - 0.0002034 s + 9.687e-06 ------------------------------------------------------------------------------------------------- s^6 + 0.07715 s^5 + 0.2502 s^4 + 0.01175 s^3 + 0.01179 s^2 + 0.0003755 s + 2.616e-05 Continuous-time transfer function.
Time=(0:1:length(IN)-1)'; %Vector de tiempo
y = lsim(G,IN,Time); % system response
figure(1)
plot(y);
hold on;
plot(IN);
plot(OUT);
legend('Estimate','Input','Output');
